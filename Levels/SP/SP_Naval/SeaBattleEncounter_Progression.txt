--function <guid>( self, event, vars, sharedvars, globalvars, deltaTime )
do
	InitVars(vars, "previousPlayerAlpha", nil)
	InitVars(vars, "deltaAck", 0)
	InitVars(vars, "shipAlpha", nil)
	InitVars(vars, "shipAngularSpeed", 0)
	InitVars(vars, "wantedDistance", 585)		-- Optimal distance to player
	InitVars(vars, "previousPlayerPos", nil)
	InitVars(vars, "targetTrail", nil)
	InitVars(vars, "destroyerAlpha", nil)
	InitVars(vars, "destroyerAngularSpeed", 0)
	InitVars(vars, "destroyerDead", false)
	InitVars(vars, "destroyerAngularSpeed", 0)
	InitVars(vars, "destroyerSunk", false)

	-- Desired distance
	local advancement = 30.0		-- Amount to advance towards target per section
	local distanceWindow = 50.0		-- Forwards and backwards window of movement (distance = 500, window 100 = span 400-600)
	local circleRadius = 1100		-- Size of area
	local acceleration = 5.0		-- m/s^2
	local deceleration = 1.5  		-- m/s^2
	local maxSpeedRatio = 1.5 		-- Max 50% faster than player
	local sidewaysWindow = 400		-- How long to the side do we keep adding distance?
	local sidewaysTolerance = 20	-- How close do we stop?
	local destroyerOffset = 250		-- How far behind is the destroyer?
	local destroyerRadius = 1250	-- Radius of destroyer circle
	local destroyerDeceleration = 10.0 -- Amount to slow down once destroyed.
	
	if event == "Advance" then
		vars.wantedDistance = vars.wantedDistance - advancement
	elseif event == "Teleport" then
		vars.shipAlpha = nil
		vars.destroyerAlpha = nil
	elseif event == "Init" then
		self.Target = LinearTransform.Identity(Vec3.FromXYZ(200, 10, 200))
		self.Destroyer = LinearTransform.Identity(Vec3.FromXYZ(100, 10, 100))
		return
	elseif event == "KillDestroyer" then
		vars.destroyerDead = true
	elseif event == "Execute" then
		deltaTime = 0.03
	elseif event == "LoadSave" then
		vars.wantedDistance = vars.wantedDistance - advancement * 11
		vars.destroyerSunk = true
		vars.destroyerDead = true
	end

	-- Check input position (sometimes we get 0,0,0 during teleports and such)
	local playerPos = self.PlayerTrans.trans;
	if playerPos.x == 0 and playerPos.z == 0 then
		return
	end

	-- Initialize target trail to player position
	if vars.targetTrail == nil then
		vars.targetTrail = {}
		for i = 1, 30 do
			vars.targetTrail[i] = playerPos
		end
	end

	-- Calculate angular coordinates for player
	local playerRadius = math.sqrt(playerPos.x ^ 2 + playerPos.z ^ 2)
	local playerAlpha = math.acos(playerPos.x / playerRadius) * sign(playerPos.z)
	
	-- Increase wanted angular distance when player is far from circle
	local playerRDist = math.min(math.abs(playerRadius - circleRadius), sidewaysWindow)
	local wantedDistance = vars.wantedDistance + math.max(playerRDist - sidewaysTolerance, 0)

	-- Calculate desired angular distance (in radians)
	local wantedAngularDistance = wantedDistance / circleRadius
	local angularDistanceWindow = distanceWindow / circleRadius

	if event == "TeleportToPlayer" then
		vars.shipAlpha = playerAlpha + wantedAngularDistance / 2
		if vars.shipAlpha > math.pi then
			vars.shipAlpha = vars.shipAlpha - 2 * math.pi
		end
		vars.destroyerAlpha = vars.shipAlpha - destroyerOffset / circleRadius
	end

	-- We also sometimes get 0-move frames - skip those
	if vars.previousPlayerAlpha == playerAlpha then
		vars.deltaAck = vars.deltaAck + deltaTime
		return
	end
	deltaTime = deltaTime + vars.deltaAck
	vars.deltaAck = 0

	-- We need to know the player's previous position to calculate angular speed
	if vars.previousPlayerAlpha == nil then
		vars.previousPlayerAlpha = playerAlpha
		vars.previousPlayerPos = playerPos
		return
	end

	-- If the angle wraps over, we need to offset the previous one full rotation
	local previousPlayerAlpha = vars.previousPlayerAlpha
	if previousPlayerAlpha > playerAlpha + math.pi then
		previousPlayerAlpha = previousPlayerAlpha - 2 * math.pi
	end
	local playerAngularSpeed = (playerAlpha - previousPlayerAlpha) / deltaTime
	vars.previousPlayerAlpha = playerAlpha
	
	-- If we're going backwards, don't turn the ship around
	if playerAngularSpeed < 0 then
		playerAngularSpeed = 0
	end

	-- Output debug stuff
	self.DebugAlpha = playerAlpha
	self.DebugR = playerRadius
	self.DebugAngularSpeed = playerAngularSpeed
	self.DebugWantedDistance = wantedAngularDistance
	self.DebugWantedLinearDistance = wantedDistance

	-- Calculate where we want the ship to be
	local wantedShipAlpha = playerAlpha + wantedAngularDistance
	if wantedShipAlpha > math.pi then
		wantedShipAlpha = wantedShipAlpha - 2 * math.pi
		if vars.shipAlpha ~= nil and vars.shipAlpha > 0 then
			vars.shipAlpha = vars.shipAlpha - 2 * math.pi
		end
	end

	-- Where is the ship? Too close, too far away?
	if vars.shipAlpha ~= nil then
		local shipTargetSpeed
		if wantedShipAlpha < vars.shipAlpha then
			-- Too far away = slower than player - Where in the window is the ship?
			local ratio = vars.shipAlpha - wantedShipAlpha
			if ratio > angularDistanceWindow then
				ratio = angularDistanceWindow
			end
			ratio = (angularDistanceWindow - ratio) / angularDistanceWindow
			shipTargetSpeed = playerAngularSpeed * ratio

		elseif wantedShipAlpha > vars.shipAlpha then
			-- Too close = faster than player - Where in the window is the ship?
			local ratio = wantedShipAlpha - vars.shipAlpha
			if ratio > angularDistanceWindow then
				ratio = angularDistanceWindow
			end
			ratio = ratio / angularDistanceWindow
			shipTargetSpeed = playerAngularSpeed + playerAngularSpeed * ratio * (maxSpeedRatio - 1)
		else
			shipTargetSpeed = playerAngularSpeed
		end

		-- Accelerate or decelerate towards target speed
		local wantedAngularSpeed = vars.shipAngularSpeed
		if shipTargetSpeed > vars.shipAngularSpeed then
			local angularAcceleration = acceleration / circleRadius
			wantedAngularSpeed = math.min(wantedAngularSpeed + angularAcceleration * deltaTime, shipTargetSpeed)
		elseif shipTargetSpeed < vars.shipAngularSpeed then
			local angularDeceleration = deceleration / circleRadius
			wantedAngularSpeed = math.max(wantedAngularSpeed - angularDeceleration * deltaTime, shipTargetSpeed)
		end
		vars.shipAngularSpeed = wantedAngularSpeed
		vars.shipAlpha = vars.shipAlpha + vars.shipAngularSpeed * deltaTime

		-- Output more debug stuff
		self.DebugCurrentPos = vars.shipAlpha
		self.DebugWantedPos = wantedShipAlpha
		self.DebugCurrentSpeed = vars.shipAngularSpeed
		self.DebugWantedSpeed = shipTargetSpeed
	else
		vars.shipAlpha = wantedShipAlpha
	end

	local shipAlpha = vars.shipAlpha
	self.DebugShipAlpha = shipAlpha
	
	-- Calculate destroyer position
	if vars.destroyerDead then
		local angularDeceleration = destroyerDeceleration / destroyerRadius
		vars.destroyerAngularSpeed = math.max(vars.destroyerAngularSpeed - angularDeceleration * deltaTime, 0)
		vars.destroyerAlpha = vars.destroyerAlpha + vars.destroyerAngularSpeed * deltaTime
	else
		vars.destroyerAngularSpeed = vars.shipAngularSpeed
		vars.destroyerAlpha = shipAlpha - destroyerOffset / circleRadius
	end

	-- Calculate ship rotation, position and output transform
	local shipRotation = LinearTransform.FromAxisY(-shipAlpha + math.pi / 2);
	local shipPosition = Vec3.FromXYZ(math.cos(shipAlpha) * circleRadius, 243.64, math.sin(shipAlpha) * circleRadius)
	self.WaspTransform = LinearTransform.FromRotTrans(shipRotation, shipPosition)

	local destroyerAlpha = vars.destroyerAlpha
	local destroyerRotation = LinearTransform.FromAxisY(-destroyerAlpha + math.pi / 2);
	local destroyerPosition = Vec3.FromXYZ(math.cos(destroyerAlpha) * destroyerRadius, 241, math.sin(destroyerAlpha) * destroyerRadius)
	if vars.destroyerSunk then
		destroyerPosition.y = destroyerPosition.y - 60
	end
	self.DestroyerTransform = LinearTransform.FromRotTrans(destroyerRotation, destroyerPosition)

	-- Check distance
	local shipAftPosition = shipPosition + shipRotation.left * 110
	local distance = Vec3.Distance(shipAftPosition, playerPos)
	self.PlayerDistance = distance
	
	-- Calculate local player to wasp space
	local playerVelocity = (playerPos - vars.previousPlayerPos) * (1 / deltaTime)
	local directedToWasp = LinearTransform.LookAtTransform(playerPos, shipPosition)
	directedToWasp.trans = directedToWasp.trans + playerVelocity * 2
	vars.previousPlayerPos = playerPos

	self.PlayerToWaspTransform = directedToWasp
	
	-- Target position is a bit laggy, so player can twist and turn to avoid being hit
	local targetPos = vars.targetTrail[1]
	for i = 1, 29 do
		vars.targetTrail[i] = vars.targetTrail[i + 1]
	end
	 -- Player will be here in 1 second = 30 frames. Aim at front of boat (7.5 m or so)
	vars.targetTrail[30] = playerPos + self.PlayerTrans.forward * 5 + playerVelocity * 1.25 + self.PlayerTrans.up * -0.5
	self.Target = LinearTransform.FromRotTrans(LinearTransform.Identity(), targetPos)	
end
