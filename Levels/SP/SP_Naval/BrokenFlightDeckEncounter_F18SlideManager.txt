	Script --function <guid>( self, event, vars, sharedvars, globalvars, deltaTime )
do
	
	-- variables
	local distance,xDist,yDist,zDist,val;
	
	if self.EntityOne ~=nil and self.EntityTwo ~=nil then 
		-- using the basic distance formula 3d version 
		--
		--			------------------------------------
		--			/(X1 - X2)^2 + (Y1-Y2)^2 + (Z1-Z2)^2
		--      -- /
		-- if you want to change to 2D check remember that y axis in fb is actually our standard z axis 
		xDist = (self.EntityOne.trans.x - self.EntityTwo.trans.x) * (self.EntityOne.trans.x - self.EntityTwo.trans.x)
		yDist = (self.EntityOne.trans.y - self.EntityTwo.trans.y) * (self.EntityOne.trans.y - self.EntityTwo.trans.y)
		zDist = (self.EntityOne.trans.z - self.EntityTwo.trans.z) * (self.EntityOne.trans.z - self.EntityTwo.trans.z)
		
		-- adding the distances
		val= xDist + yDist + zDist
		-- getting the square root to get the distance
		distance = math.sqrt(val)
		
		-- ouput the distance so we can see it for debug
		
		self.Distance = distance
		
		-- check whether we are in range or not
		if distance <= 2.5 then
			self.FireEvent("OnWithin")
		else 
			self.FireEvent("OnOutside")
		end 
	end
end

		--			------------------------------------

		-- if you want to change to 2D check remember that y axis in fb is actually our standard z axis
